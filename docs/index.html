soup = BeautifulSoup(r.text, "html.parser")

tables = soup.find_all("table")
best, score_best = None, -1
for tbl in tables:
    headers = [th.get_text(" ", strip=True).lower() for th in tbl.find_all("th")]
    score = sum(k in " ".join(headers) for k in ["region", "start", "maximum", "end"])
    if score > score_best:
        best, score_best = tbl, score
rows = []
for tr in best.select("tbody tr"):
    tds = tr.find_all("td")
    if len(tds) < 8:
        continue
    rows.append({
        "rank": tds[0].get_text(strip=True),
        "class": tds[1].get_text(strip=True),
        "date": tds[2].get_text(strip=True),
        "region": tds[3].get_text(strip=True),
        "starttime": tds[4].get_text(strip=True),
        "Maxtime": tds[5].get_text(strip=True),
        "end time": tds[6].get_text(strip=True),
        "movie": tds[7].get_text(strip=True),
    })

df_swl = pd.DataFrame(rows)
df_swl["rank"] = pd.to_numeric(df_swl["rank"], errors="coerce").astype("Int64")

for c in ["class", "date", "region", "starttime", "Maxtime", "end time", "movie"]:
    df_swl[c] = df_swl[c].astype(str).str.strip()

Question 1:Replication

(a) To replicate the SpaceWeatherLive list, I converted the flare classes (like M1.3, X2, etc.) into numeric values and sorted them by strength. This way I could pull out the top 50 strongest flares, which lines up well with how SpaceWeatherLive ranks them.

(b) My table also looks similar to theirs, with columns like region, start time, and end time. The main difference is that instead of showing the flares maximum time, I used the Coronal mass ejection time from the NASA data.

import re
import pandas as pd
import numpy as np

df_flares = df.copy()
def flare_to_numeric(val):
    if not isinstance(val, str):
        return np.nan
    s = val.strip().upper()
    m = re.match(r"^([ABCMX])\s*([0-9]+(?:\.[0-9]+)?)", s)
    if not m:
        return np.nan
    letter = m.group(1)
    mag = float(m.group(2))
    scale = {"A": 1e-4, "B": 1e-3, "C": 1e-2, "M": 1e-1, "X": 1.0}
    return mag * scale[letter]
df_flares["flare_strength"] = df_flares["Xray"].apply(flare_to_numeric)
df_top50 = (
    df_flares.dropna(subset=["flare_strength"])
             .sort_values(["flare_strength", "start_datetime"], ascending=[False, True])
             .head(50)
             .reset_index(drop=True)
)
df_top50.insert(0, "rank", df_top50.index + 1)

Question 2: Integration

a) I matched flares between the NASA dataset and the SpaceWeatherLive dataset using the start date only, ignoring the exact time. This is because the recorded times for start/peak/end often differ slightly between catalogs, but the day of the flare is consistent

b)I also used the NOAA region number (last four digits) in the NASA dataset as a unique identifier for the active region that produced the flare. Both datasets include this information, and its usually stable across catalogs, so it makes the match much more reliable.

c)By combining date and region, I joined the datasets and added a new column called swl_rank to the NASA top-50 flares. This column shows where that same flare ranks in the SWL top-50 list.
